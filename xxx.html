      padding: 15px 25px;

      box-shadow: 0px 4px 6px rgba(0,0,0,0.2);

    }

    .btn:active {

      background: #ad1457;

    }

  </style>

  <script src="https://cdn.jsdelivr.net/npm/phaser@3.60.0/dist/phaser.js"></script>

<body>

  <h2>🎉 Catch 26 Hearts for Your Birthday Surprise 🎉</h2>

  <div id="game"></div>

 

  <!-- Tombol sentuh untuk HP -->

  <div class="touch-controls">

    <button class="btn" id="leftBtn">⬅️</button>

    <button class="btn" id="rightBtn">➡️</button>

  </div>

 

  <script>

    const config = {

      type: Phaser.AUTO,

      width: 400,

      height: 600,

      parent: "game",

      physics: {

        default: "arcade",

        arcade: { gravity: { y: 200 }, debug: false }

      },

      scene: { preload, create, update }

    };

 

    let player, hearts, cursors, music;

    let score = 0;

    let scoreText;

    let moveLeft = false;

    let moveRight = false;

 

    const game = new Phaser.Game(config);

 

    function preload() {

      this.load.image("player", "https://i.ibb.co/p1cLkR0/player.png");

      this.load.image("heart", "https://i.ibb.co/k5m2P2h/heart.png");

      this.load.image("bg", "https://i.ibb.co/X8wPCkq/bg.jpg");

      this.load.audio("bgmusic", "https://cdn.pixabay.com/download/audio/2022/03/15/audio_bf8b95f1a1.mp3?filename=romantic-music-11085.mp3");

    }

 

    function create() {

      this.add.image(200, 300, "bg").setScale(0.5);

 

      player = this.physics.add.image(200, 550, "player").setScale(0.3);

      player.setCollideWorldBounds(true);

 

      hearts = this.physics.add.group({

        key: "heart",

        repeat: 0,

        setXY: { x: Phaser.Math.Between(50, 350), y: 0 }

      });

 

      this.physics.add.collider(player, hearts, collectHeart, null, this);

 

      cursors = this.input.keyboard.createCursorKeys();

 

      scoreText = this.add.text(10, 10, "Score: 0 / 26", {

        fontSize: "20px",

        fill: "#000"

      });

 

      // Musik romantis

      music = this.sound.add("bgmusic", { loop: true, volume: 0.5 });

      music.play();

 

      // Kontrol tombol sentuh

      document.getElementById("leftBtn").addEventListener("touchstart", () => moveLeft = true);

      document.getElementById("leftBtn").addEventListener("touchend", () => moveLeft = false);

      document.getElementById("rightBtn").addEventListener("touchstart", () => moveRight = true);

      document.getElementById("rightBtn").addEventListener("touchend", () => moveRight = false);

    }

 

    function update() {

      if (cursors.left.isDown || moveLeft) {

        player.setVelocityX(-200);

      } else if (cursors.right.isDown || moveRight) {

        player.setVelocityX(200);

      } else {

        player.setVelocityX(0);

      }

 

      // Respawn hearts jika jatuh

      hearts.children.iterate(function (child) {

        if (child.y > 600) {

          child.y = 0;

          child.x = Phaser.Math.Between(50, 350);

        }

      });

    }

 

    function collectHeart(player, heart) {

      heart.y = 0;

      heart.x = Phaser.Math.Between(50, 350);

      score++;

      scoreText.setText("Score: " + score + " / 26");

 

      if (score >= 26) {

        endGame(this);

      }

    }

 

    function endGame(scene) {

      scene.physics.pause();

      player.setTint(0xff69b4);

 

      scene.add.text(50, 250, "🎂 Happy 26th Birthday! 🎂", {

        fontSize: "24px",

        fill: "#e91e63",

        fontStyle: "bold"

      });

 

      scene.add.text(70, 300, "I love you forever 💖", {

        fontSize: "18px",

        fill: "#880e4f"

      });

    }

  </script>
